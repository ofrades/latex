\textit{Como na Missa Me exspectavérunt, página \pageref{virgensmartires2}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} ómnium largítor bonórum, qui in fámula tua Bibiána cum virginitátis flore martýrii palmam conjunxísti: mentes nostras ejus intercessióne tibi caritáte conjúnge; ut, amótis perículis, prǽmia consequámur ætérna. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, dispensador de todos os bens, que unistes a flor da virgindade à palma do martírio na pessoa de vossa serva Bibiana, dignai-Vos por sua intercessão unir pela caridade as nossas almas convosco, a fim de que, afastados todos os perigos, alcancemos os prémios eternos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
