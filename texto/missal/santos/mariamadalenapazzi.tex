\subsectioninfo{S. Maria Madalena de Pazzi, Virgem}{29 de Maio}

\textit{Como na Missa Dilexísti justitiam, página \pageref{virgemnaomartir1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} virginitátis amátor, qui beátam Maríam Magdalénam Vírginem, tuo amóre succénsam, cœléstibus donis decorásti: da; ut, quam festíva celebritáte venerámur, puritáte et caritáte imitémur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que amais a virgindade e ornastes com dons celestiais a B. Maria Madalena, vossa Virgem, cujo coração estava abrasado em amor por Vós, concedei-nos que, honrando-a nós com a celebração desta festa, a imitemos também na sua pureza e caridade. Por nosso Senhor \emph{\&c.}
}\end{paracol}
