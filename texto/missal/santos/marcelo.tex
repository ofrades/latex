\textit{Como na Missa Si díligis me, página \pageref{sumospontifices}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{P}{reces} pópuli tui, quǽsumus, Dómine, cleménter exáudi: ut beáti Marcélli Mártyris tui atque Pontíficis méritis adjuvémur, cujus passióne lætámur. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} dignai-Vos ouvir clementemente as preces do vosso povo, a fim de que sejamos auxiliados pelos méritos do B. Marcelo, vosso Pontífice e Mártir, cujo martírio festejamos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
