\subsectioninfo{S. Vicente Ferrer, Conf.}{5 de Abril}

\textit{Como na Missa Os justi, página \pageref{confessoresnaopontifices1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui Ecclésiam tuam beáti Vincéntii Confessóris tui méritis et prædicatióne illustráre dignátus es: concéde nobis fámulis tuis; ut et ipsíus instruámur exémplis et ab ómnibus ejus patrocínio liberémur advérsis. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que Vos dignastes ilustrar a vossa Igreja com os méritos e a pregação do B. Vicente, vosso Confessor, concedei a estes vossos servos a graça de serem instruídos com seus exemplos e livres de todas as adversidades pela sua protecção. Por nosso Senhor \emph{\&c.}
}\end{paracol}
