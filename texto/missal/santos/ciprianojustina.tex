\subsectioninfo{S. S. Cipriano e Justina, Mártires}{26 de Setembro}

\textit{Como na Missa Salus autem, página \pageref{muitosmartires3}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{B}{eatórum} Mártyrum Cypriáni et Justínæ nos, Dómine, fóveant continuáta pæsídia: quia non désinis propítius intuéri, quos tálibus auxíliis concésseris adjuvári. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{F}{azei,} Senhor, que incessantemente nos alegremos com a protecção dos vossos B. B. Mártires Cipriano e Justina, visto que não podeis deixar de acolher benignamente aqueles a quem concedeis a assistência de tais auxílios. Por nosso Senhor \emph{\&c.}
}\end{paracol}
