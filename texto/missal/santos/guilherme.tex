\subsectioninfo{S. Guilherme, Abade}{25 de Junho}

\textit{Como na Missa Os justi, página \pageref{abades}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui infirmitáti nostræ ad teréndam salútis viam in Sanctis tuis exémplum et præsídium collocásti: da nobis, ita beáti Guliélmi Abbátis mérita venerári; ut ejúsdem excipiámus suffrágia et vestígia prosequámur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que para facilitar à nossa fraqueza o caminho da salvação no auxiliais com o exemplo e com a assistência dos vossos Santos, concedei-nos a graça de imitar os méritos do B. Abade Guilherme de modo que mereçamos o socorro das suas preces e sigamos os seus passos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
