\subsectioninfo{Trasladação de S. António de Lisboa}{15 de Fevereiro}

\textit{Como na Missa de Santo António de Lisboa, a 13 de Junho, página \pageref{antoniolisboa}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{S}{ancti} Confessórisóris tui Antónii tribue nos, Dómine, intercessióne adjuvári: ut cujus Translatiónem ánnua celebitáte recólimus; ejus apud te précibus et méritis, post hujus vitæ exsílium, ad consórtium transférri merámur Beatórum. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{C}{oncedei-nos,} Senhor, que sejamos auxiliados pela intercessão do vosso Santo Confessor António, a fim de que, celebrando anualmente a sua Trasladação, por suas preces e méritos junto de Vós, mereçamos ser levados, após o exílio desta vida, ao consórcio dos bem-aventurados. Por nosso Senhor \emph{\&c.}
}\end{paracol}
