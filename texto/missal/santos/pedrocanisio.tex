\subsectioninfo{S. Pedro Canísio, Conf. e Doutor}{27 de Abril}

\textit{Como na Missa In médio Ecclésiae, página \pageref{doutores}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui ad tuéndam cathólicam iidem beátum Petrum Confessórem tuum virtúte et doctrína roborásti: concéde propítius; ut ejus exémplis et mónitis errántes ad salútem resipíscant, et fidéles in veritátis confessióne persevérent. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que revestistes o B. Pedro, vosso Confessor, com os dons da virtude e da ciência para a defesa da fé católica, concedei-nos propício que seus exemplos e ensinos atraiam ao caminho da salvação aqueles que caíram no erro e façam perseverar os fiéis na confissão da verdade. Por nosso Senhor \emph{\&c.}
}\end{paracol}
