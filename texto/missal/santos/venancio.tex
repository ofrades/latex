\subsectioninfo{S. Venâncio, Mártir}{18 de Maio}

\textit{Como na Missa Protexísti me, página \pageref{martir}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui hunc diem beáfi Venántii Martyris tui triúmpho consecrásti: exáudi preces pópuli tui et præsta: ut, qui ejus mérita venerámur, fídei constántiam imitémur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que consagrastes este dia com o triunfo do B. Venâncio, vosso Mártir, ouvi as preces do vosso povo, e permiti que, honrando os seus méritos, imitemos também a constância da sua fé. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{H}{anc} oblatiónem, omnípotens Deus, beáti Venántii mérita tibi reddant accéptam: ut, ipsíus subsidiis adjuti, glóriæ ejus consortes efficiámur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} esta oferta, ó Deus omnipotente, Vos seja agradável pelos méritos do B. Venâncio, a fim de que, assistidos com seu auxílio, nos tornemos participantes da sua glória. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{S}{úmpsimus,} Dómine, ætérnæ vitæ sacraménta, te humiliter deprecántes: ut, beáto Venántio Mártyre tuo pro nobis deprecánte, véniam nobis concílient et grátiam. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} havendo recebido os sacramentos da vida eterna, humildemente Vos suplicamos que as preces do B. Venâncio, vosso Mártir, nos alcancem o perdão e a graça. Por nosso Senhor \emph{\&c.}
}\end{paracol}
