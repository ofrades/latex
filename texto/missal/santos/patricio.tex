\subsectioninfo{S. Patrício, B. e Conf.}{17 de Março}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui ad prædicándam géntibus glóriam tuam beátum Patrícium Confessórem atque Pontíficem míttere dignátus es: ejus méritis et intercessióne concéde; ut, quæ nobis agénda prǽcipis, te miseránte adimplére póssimus. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que Vos dignastes mandar B. Patrício, Confessor e Bispo, anunciar a vossa glória aos gentios, concedei-nos pelos seus méritos e intercessão que com o socorro da vossa misericórdia possamos cumprir o que nos ordenais. Por nosso Senhor \emph{\&c.}
}\end{paracol}
