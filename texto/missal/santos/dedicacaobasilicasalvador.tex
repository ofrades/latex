\subsectioninfo{Dedicação da Basílica do Salvador}{9 de Novembro}

\textit{Como na Missa Terríbilis est, página \pageref{dedicacaoigreja}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui nos beáti Theodóri Mártyris tui confessióne gloriósa circúmdas et prótegis: præsta nobis ex ejus imitatióne profícere et oratióne fulcíri. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que nos guardais e protegeis pela gloriosa confissão do B. Teodoro, vosso Mártir, concedei-nos a graça de aproveitarmos com seus exemplos e de sermos auxiliados com suas orações. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{S}{uscipe,} Dómine, fidélium preces cum oblatiónibus hostiárum: et, intercedénte beáto Theodóro Mártyre tuo, per hæc piæ devotiónis offícia ad cœléstem glóriam transeámus. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{R}{ecebei,} Senhor, as preces dos fiéis juntamente com a oblata destas hóstias, a fim de que, pela intercessão do B. Teodoro, vosso Mártir, obtenhamos, em virtude destes mystérios de pia devoção, a graça de sermos conduzidos até à glória celestial. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{P}{ræsta} nobis, quǽsumus, Dómine: intercedénte beáto Theodóro Mártyre tuo; ut, quod ore contíngimus, pura mente capiámus. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{P}{ermiti,} Senhor, pela intercessão do B. Teodoro, vosso Mártir, que conservemos com o coração puro o sacramento que a nossa boca recebeu. Por nosso Senhor \emph{\&c.}
}\end{paracol}
