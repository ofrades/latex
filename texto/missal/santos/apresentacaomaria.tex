\subsectioninfo{Apresentação da B. V. Maria}{21 de Novembro}

\textit{Como na Missa Salve, sancta Parens, página \pageref{comumfestasmaria1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui beátam Maríam semper Vírginem, Spíritus Sancti habitáculum, hodiérna die in templo præsentári voluísti: præsta, quǽsumus; ut, ejus intercessióne, in templo glóriæ tuæ præsentári mereámur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que quisestes que a B. V. Maria, em quem habitava o Espírito Santo, fosse neste dia apresentada no templo, permiti pela sua intercessão que mereçamos ser apresentados no templo da vossa glória. Por nosso Senhor \emph{\&c.}
}\end{paracol}
