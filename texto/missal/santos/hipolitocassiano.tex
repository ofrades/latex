\subsectioninfo{S. S. Hipólito e Cassiano, Mártires}{13 de Agosto}

\textit{Como na Missa Salus autem, página \pageref{muitosmartires3}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{a,} quǽsumus, omnípotens Deus: ut beatórum Mártyrum tuórum Hippolýti et Cassiáni veneránda sollémnitas, et devotiónem nobis áugeat et salútem. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{C}{oncedei-nos,} ó Deus omnipotente, Vos rogamos, que a veneranda festa dos vossos B. B. Mártires Hipólito e Cassiano aumente a nossa piedade e nos faça alcançar a salvação. Por n \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{R}{éspice,} Dómine, múnera pópuli tui, Sanctórum festivitáte votíva: et tuæ testificátio veritátis nobis profíciat ad salútem. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{D}{ignai-Vos} olhar, Senhor, para as oblatas que o vosso povo Vos oferece nesta festa dos vossos Santos, e fazei que o testemunho, por eles dado à vossa verdade, nos seja útil para a salvação. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{S}{acramentórum} tuórum, Dómine, commúnio sumpta nos salvet: et in tuæ veritátis luce confírmet. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} a comunhão dos vossos sacramentos, que recebemos, Senhor, nos alcance a salvação e nos confirme na profissão da vossa doutrina. Por nosso Senhor \emph{\&c.}
}\end{paracol}
