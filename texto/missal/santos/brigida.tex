\subsectioninfo{Santa Brígida, Viúva}{8 de Outubro}

\textit{Como na Missa Cognóvi, Dómine, página \pageref{nemvirgensnemmartires}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{ómine,} Deus noster, qui beátæ Birgíttæ per Fílium tuum unigénitum secreta cœléstia revelásti: ipsíus pia intercessióne da nobis, fámulis tuis; in revelatióne sempitérna glóriæ tuæ gaudére lætántes. Per eúndem Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} nosso Deus, que pelo vosso Unigénito Filho revelastes os segredos celestiais à B. Brígida, concedei-nos, já que somos vossos servos, que nos alegremos na felicidade da contemplação da vossa eterna glória. Por nosso Senhor \emph{\&c.}
}\end{paracol}
