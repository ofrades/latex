\subsectioninfo{S. Vicente de Paulo, Conf.}{19 de Julho}

\textit{Como Missa Justus ut palma, página \pageref{confessoresnaopontifices2}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui, ad evangelizándum paupéribus et ecclesiástici órdinis decórem promovéndum, beátum Vincéntium apostólica virtúte roborásti: præsta, quǽsumus; ut, cujus pia mérita venerámur, virtútum quoque instruámur exémplis. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que ornastes o B. Vicente com a coragem e as virtudes apostólicas para a
evangelização dos pobres e para promover o decoro da ordem eclesiástica, permiti, Vos imploramos, que, honrando os seus piedosos méritos, sejamos instruídos com o exemplo de suas virtudes. Por nosso Senhor \emph{\&c.}
}\end{paracol}
