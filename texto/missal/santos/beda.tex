\subsectioninfo{S. Beda, Conf. e Doutor}{27 de Maio}

\textit{Como na Missa In médio Ecclésiae, página \pageref{doutores}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui Ecclésiam tuam beáti Bedæ Confessóris tui atque Doctóris eruditióne claríficas: concéde propítius fámulis tuis; ejus semper ilustrári sapiéntia et méritis adjuvári. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que ilustrais a vossa Igreja com a erudição do B. Beda, vosso Confessor e Doutor, concedei propício aos vossos servos que sejam sempre ilustrados pela sua doutrina e socorridos pelos seus méritos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
