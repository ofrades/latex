\subsectioninfo{S. Casimiro, Conf.}{4 de Março}

\textit{Como na Missa Os justi, página \pageref{confessoresnaopontifices1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui inter regáles delicias et mundi illécebras sanctum Casimírum virtúte constantiæ roborásti: quǽsumus; ut ejus intercessióne fidéles tui terréna despíciant, et ad cœléstia semper aspírent. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que no meio das delícias régias e dos atractivos do mundo fortalecestes S. Casimiro com a virtude da constância, fazei, Vos rogamos, que por sua intercessão os fiéis desprezem os bens terrenos e aspirem sempre aos celestiais. Por nosso Senhor \emph{\&c.}
}\end{paracol}
