\subsectioninfo{S. S. Sérgio, Baco e Outros, Mártires}{7 de Outubro}

\textit{Como na Missa Sapiéntiam sanctórum, página \pageref{muitosmartires2}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{S}{anctórum} Martyrum tuórum nos, Dómine, Sérgii, Bacchi, Marcélli et Apuléji beáta mérita prosequántur: et tuo semper fáciant amóre fervéntes. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{F}{azei,} Senhor, que os méritos dos vossos Santos Mártires Sérgio, Baco, Marcelo e Apúlio nos acompanhem e nos comuniquem incessantemente o fervor do vosso amor. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{M}{ajestátem} tuam nobis, Dómine, quǽsumus, hæc hóstia reddat immolánda placátam, tuórum digna postulatióne Sanctórum. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} esta hóstia, que vai ser imolada em vosso louvor, Vos rogamos, Senhor, possa pelas dignas preces dos vossos Santos aplacar a vossa majestade. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{S}{acraméntis,} Dómine, muniámur accéptis: et sanctórum Mártyrum tuórum Sérgii, Bacchi, Marcélli et Apuléji intercessióne, contra omnes nequítias irruéntes, annis cœléstibus protegámur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} os sacramentos que recebemos, Senhor, nos confortem; e que pela intercessão dos vossos Santos Sérgio, Baco, Marcelo e Apúlio sejamos fortalecidos com as armas celestiais e defendidos contra todos os ataques dos espíritos malignos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
