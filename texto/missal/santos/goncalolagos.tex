\subsectioninfo{S. Gonçalo de Lagos}{27 de Outubro}

\textit{Como na Missa Os justi, página \pageref{confessoresnaopontifices1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus} virtútem, qui beátum Gundisálvum Confessórem tuum ánimi demissióne, morum candóre, ac singuláre caritáte mirábilem effecísti: concéde, quæsumus; ut ejus exémpla imitántes in terris, coronári cum ipso merámur in cælis. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus das virtudes, que tornastes admirável na humildade de espírito, na candura de Costumes e na singular caridade o B. Gonçalo, vosso Confessor, concedei-nos, Vos imploramos, que, Imitando seus exemplos na terra, mereçamos com ele ser coroados nos céus. Por nosso Senhor \emph{\&c.}
}\end{paracol}
