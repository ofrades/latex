\subsectioninfo{Santa Isabel, Viúva}{19 de Novembro}

\textit{Como na Missa Cognóvi, Dómine, página \pageref{nemvirgensnemmartires}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{T}{uórum} corda fidélium, Deus miserátor, illústra: et, beátæ Elisabeth précibus gloriósis; fac nos próspera mundi despícere, et cœlésti semper consolatióne gaudére. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus de misericórdia, esclarecei os corações dos vossos fiéis; e, pelas gloriosas preces da B. Isabel, concedei-nos a graça de desprezarmos as prosperidades deste mundo e de gozarmos perpetuamente a consolação celestial. Por nosso Senhor \emph{\&c.}
}\end{paracol}
