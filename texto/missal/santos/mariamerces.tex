\subsectioninfo{B. V. Maria das Mercês}{24 de Setembro}

\textit{Como na Missa Salve, sancta Parens da Virgem Maria, página \pageref{comumfestasmaria1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui per gloriosíssimam Fílii tui Matrem, ad liberandos Christi fidéles a potestáte paganórum, nova Ecclésiam tuam prole amplificáre dignátus es: præsta, quǽsumus; ut, quam pie venerámur tanti óperis institutrícem, ejus páriter méritis et intercessióne, a peccátis ómnibus et captivitáte dǽmonis liberémur. Per eúndem Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que pela gloriosíssima Mãe do vosso Filho Vos dignastes aumentar a vossa Igreja com uma nova família, destinada a libertar os fiéis de Cristo do poder dos pagãos, concedei-nos, Vos rogamos, que, venerando piedosamente a inspiradora de uma tão notável instituição, sejamos, graças aos seus méritos e intercessão, livres de todos os pecados e do cativeiro do demónio. Pelo mesmo \emph{\&c.}
}\end{paracol}
