\subsectioninfo{S. S. Proto e Jacinto, Mártires}{11 de Setembro}

\textit{Como na Missa Salus autem, página \pageref{muitosmartires3}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{B}{eatórum} Mártyrum tuórum Proti et Hyacínthi nos, Dómine, fóveat pretiósa conféssio: et pia júgiter intercéssio tueátur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} a preciosa confissão da fé dos vossos B. B. Mártires Proto e Jacinto nos anime, e que eles nos protejam incessantemente com os méritos da sua piedosa intercessão. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{P}{ro} sanctórum Martyrum tuórum Proti et Hyacínthi commemoratióne, múnera tibi, Dómine, quæ debémus, exsólvimus: præsta, quǽsumus; ut remédium nobis perpétuæ salútis operéntur. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{V}{os} oferecemos, Senhor, em memória dos vossos Santos Mártires Proto e Jacinto, estas oblatas, que Vos são devidas, e permiti, Vos rogamos, que nos curem e proporcionem o remédio da salvação. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{U}{t} percépta nos, Dómine, tua sancta puríficent: beatórum Mártyrum tuórum Proti et Hyacínthi, quǽsumus, ímploret orátio. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{V}{os} pedimos, Senhor, fazei que os mistérios agora recebidos nos alcancem, pelas preces dos vossos B. B. Mártires Proto e Jacinto, a graça de sermos purificados. Por nosso Senhor \emph{\&c.}
}\end{paracol}
