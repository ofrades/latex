\subsectioninfo{Santa Tecla, Virgem e Mártir}{23 de Setembro}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{a,} quǽsumus, omnípotens Deus: ut, qui beátæ Theclæ Vírginis et Mártyris tuæ natalítia cólimus; et ánnua sollemnitáte lætémur, et tantæ fídei proficiámus exémplo. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus omnipotente, celebrando nós o nascimento da B. Tecla, vossa Virgem e Mártir, concedei-nos, Vos suplicamos, que nos alegremos com sua festa anual e nos sejam proveitosos os exemplos da sua grande fé. Por nosso Senhor \emph{\&c.}
}\end{paracol}
