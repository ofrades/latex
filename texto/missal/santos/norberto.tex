\subsectioninfo{S. Norberto, B e Conf.}{6 de Junho)}

\textit{Como na Missa Státuit ei, página \pageref{confessorespontifices1}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui beátum Norbértum Confessórem tuum atque Pontíficem verbi tui præcónem exímium effecísti, et per eum Ecclésiam tuam nova prole fœcundásti: præsta, quǽsumus; ut, ejúsdem suffragántibus méritis, quod ore simul et ópere dócuit, te adjuvánte, exercére valeámus. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que dispusestes que o B. Norberto, vosso Confessor e Pontífice, fosse exímio pregador da vossa palavra, e que por ele destes à vossa Igreja uma nova família, concedei-nos, Vos imploramos, que, auxiliados com seus méritos, possamos com vossa assistência pôr em prática tudo quanto ensinou com suas palavras e acções. Por nosso Senhor \emph{\&c.}
}\end{paracol}
