\subsectioninfo{S. Saturnino, Mártir}{29 de Novembro}

\textit{Como na Missa Lætábitur justus, página \pageref{martirnaopontifice2}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} qui nos beáti Saturníni Martyris tui concédis natalítia pérfrui: ejus nos tríbue méritis adjuvári. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que nos permitistes a alegria de celebrarmos o aniversário do nascimento do vosso B. Mártir Saturnino, concedei-nos ainda que sejamos auxiliados pelos seus méritos. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{M}{únera,} Dómine, tibi dicáta sanctífica: et, intercedénte beáto Saturníno Mártyre tuo, per hæc eádem nos placátus inténde. Per Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{antificai,} Senhor, estas ofertas, que Vos dedicamos, e por intercessão do B. Saturnino, vosso Mártir, olhai aplacado para nós. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{S}{anctíflcet} nos, quǽsumus, Dómine, tui percéptio sacraménti: et intercessióne Sanctórum tuórum tibi reddat accéptos. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} Vos suplicamos, permiti que a recepção do vosso sacramento nos santifique e que por intercessão dos vossos santos nos alcance a graça de Vos agradarmos. Por nosso Senhor \emph{\&c.}
}\end{paracol}
