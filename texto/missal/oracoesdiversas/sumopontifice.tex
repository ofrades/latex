\subsection{4.ª Pelo Sumo Pontífice}\label{sumopontifice}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} ómnium fidélium pastor et rector, fámulum tuum {\redx N.}, quem pastórem Ecclésiæ tuæ præésse voluísti, propítius réspice: da ei, quǽsumus, verbo et exémplo, quibus præest, profícere; ut ad vitam, una cum grege sibi crédito, pervéniat sempitérnam. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, pastor e guia de todos os fiéis, olhai propício para o vosso servo {\redx N.}, que constituístes Pastor da vossa Igreja; concedei-lhe, Vos suplicamos, a graça de edificar os seus súbditos com suas palavras e exemplos, a fim de que com o rebanho, que lhe foi confiado, alcance a vida eterna. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{O}{blátis,} quǽsumus, Dómine, placáre munéribus: et fámulum tuum {\redx N.}, quem pastórem Ecclésiæ tuæ præésse voluísti, assídua protectióne gubérna. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{D}{eixai-Vos} aplacar, Senhor, Vos suplicamos, com a oferta destes dons; e governai com vossa contínua protecção o vosso servo {\redx N.}, que constituístes Pastor da vossa Igreja. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{H}{æc} nos, quǽsumus, Dómine, divíni sacraménti percéptio prótegat: et fámulum tuum {\redx N.}, quem pastórem Ecclésiæ tuæ præésse voluísti; una cum commísso sibi grege, salvet semper et múniat. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} Vos rogamos, permiti que a comunhão deste divino sacramento nos proteja; e salve e defenda, com o rebanho que lhe foi confiado, o vosso servo {\redx N.}, que constituístes Pastor da vossa Igreja. Por nosso Senhor \emph{\&c.}
}\end{paracol}
