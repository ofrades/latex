\subsection{6.ª Pelo Rei}\label{rei}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\qlettrine{Q}{uǽsumus,} omnípotens Deus: ut fámulus tuus {\redx N.}, rex noster, qui tua miseratióne suscépit regni gubernácula, virtútum etiam ómnium percípiat increménta; quibus decénter ornátus, et vitiórum monstra devitáre et ad te, qui via, véritas et vita es, gratiósus váleat perveníre. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{V}{os} suplicamos, ó omnipotente Deus, permiti que o vosso servo {\redx N.}, nosso rei, o qual, por efeito da vossa misericórdia, recebeu as rédeas do governo, alcance também incremento em todas as virtudes, a fim de que, sendo, como convém, ornado com elas, possa evitar o horror dos vícios e, cheio de merecimentos, se aproxime de Vós, que sois o caminho, a verdade e a vida. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{M}{únera,} Dómine, quǽsumus, oblata sanctífica: ut et nobis Unigéniti tui Corpus et Sanguis fiant; ei regi nostro ad obtinéndam ánimæ corporísque salútem, et peragéndum injúnctum offícium, te largiénte, usquéquaque profíciant. Per eúndem Dóminum nostrum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{antificai,} Senhor, Vos imploramos, estas oblatas que Vos são oferecidas, a fim de que, para nosso proveito, se convertam no Corpo e no Sangue de vosso Filho Unigénito; e que, pela vossa misericórdia, sejam úteis ao nosso rei, para lhe alcançarem a salvação da alma e do corpo e o auxiliarem a desempenhar o cargo de que foi investido. Pelo mesmo nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{H}{æc,} Dómine, oblátio salutáris fámulum tuum {\redx N.}, regem nostrum, ab ómnibus tueátur advérsis: quaténus et ecclesiásticæ pacis obtíneat tranquillitátem; et, post hujus témporis decúrsum, ad ætérnam pervéniat hereditátem. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} que esta salutar oblação proteja o vosso servo {\redx N.}, nosso rei, contra todas as adversidades, a fim de que alcance a tranquilidade, que proporciona a paz da Igreja, e, após esta vida, consiga a herança eterna. Por nosso Senhor \emph{\&c.}
}\end{paracol}
