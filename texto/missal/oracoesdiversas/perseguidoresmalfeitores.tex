\subsection{11.ª Contra os nossos perseguidores e malfeitores}\label{perseguidoresmalfeitores}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{H}{óstium} nostrórum, quǽsumus, Dómine, elíde supérbiam: et eórum contumáciam déxteræ tuæ virtúte prostérne. Per Dóminum nostrum. \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{E}{smagai,} Senhor, o orgulho dos nossos inimigos, e, Vos suplicamos, com o poder da vossa dextra, reprimi a sua arrogância. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{H}{ujus,} Dómine, virtúte mystérii, et a própriis mundémur occúltis, et ab inimicórum liberémur insídiis. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\rlettrine{S}{enhor,} que pela virtude deste mystério sejamos purificados das nossas manchas ocultas e livres das ciladas dos nossos inimigos. Por nosso Senhor \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\qlettrine{P}{rotéctor} noster, áspice. Deus, et ab inimicórum nos defénde perículis: ut, omni perturbatióne submóta, líberis tibi méntibus serviámus. Per Dóminum \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, que sois o nosso protector, dignai-Vos lançar sobre nós os vossos olhares, livrando-nos dos perigos com que nos afligem os nossos inimigos, a fim de que, sendo afastados todos os motivos de perturbação, nos empreguemos no vosso serviço, livres de todos os cuidados. Por nosso Senhor \emph{\&c.}
}\end{paracol}
