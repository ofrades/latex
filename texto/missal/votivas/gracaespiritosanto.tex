\subsection{Suplicando a Graça do Espírito Santo}

\textit{Como na Missa do Espírito Santo, página \pageref{votivaespiritosanto}, excepto:}

\paragraph{Oração}
\begin{paracol}{2}\latim{
\rlettrine{D}{eus,} cui omne cor patet, et omnis volúntas lóquitur, et quem nullum latet secrétum: purífica per infusiónem Sancti Spíritus cogitatiónes cordis nostri; ut te perfécte dilígere et digne laudáre mereámur. Per Dóminum \emph{\&c.} in unitáte ejúsdem Spíritus Sancti \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} Deus, de quem é conhecido o íntimo de todo o coração e os desejos de toda a vontade, assim como não ignorais nenhum segredo, purificai, pela difusão do Espírito Santo, os pensamentos do nosso coração, a fim de que Vos amemos com perfeição e mereçamos louvar-Vos dignamente. Por nosso Senhor \emph{\&c.} em unidade do mesmo Espírito Santo \emph{\&c.}
}\end{paracol}

\paragraph{Secreta}
\begin{paracol}{2}\latim{
\rlettrine{H}{æc} oblátio, quǽsumus, Dómine, cordis nostri máculas emúndet: ut Sancti Spíritus digna efficiátur habitátio. Per Dóminum \emph{\&c.} in unitáte ejúsdem Spíritus Sancti \emph{\&c.}
}\switchcolumn\portugues{
\qlettrine{Q}{ue} esta oblação, Senhor, Vos suplicamos, apague as manchas do nosso coração, a fim de que ele se torna em digno habitáculo do Espírito Santo. Por nosso Senhor \emph{\&c.} em unidade do mesmo Espírito Santo \emph{\&c.}
}\end{paracol}

\paragraph{Postcomúnio}
\begin{paracol}{2}\latim{
\rlettrine{C}{oncéde,} quǽsumus, omnípotens Deus, Sanctum nos Spíritum votis promeréri sédulis: quaténus, ejus grátia, et ab ómnibus liberémur tentatiónibus, et peccatórum nostrórum indulgéntiam percípere mereámur. Per Dóminum \emph{\&c.} in unitáte ejúsdem Spíritus Sancti \emph{\&c.}
}\switchcolumn\portugues{
\slettrine{Ó}{} omnipotente Deus, Vos suplicamos, concedei-nos que pelos nossos votos nos tornemos dignos de o Espírito Santo habitar em nós, de modo que pela sua graça sejamos livres de todas as tentações e mereçamos alcançar o perdão dos nossos pecados. Por nosso Senhor \emph{\&c.} em unidade do mesmo Espírito Santo \emph{\&c.}
}\end{paracol}
